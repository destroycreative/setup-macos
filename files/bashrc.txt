### Support local aliases
export PROMPT_COMMAND=prompt_command
alias rla='bashrc=""'
function prompt_command() {
  if [[ "$bashrc" != "$PWD" && "$PWD" != "$HOME" && -e .bashrc ]]; then
    bashrc="$PWD";
    source .bashrc;
  fi;
}
